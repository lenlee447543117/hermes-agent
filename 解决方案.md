```python
    global current_task
    if current_task["status"] == "executing":

    current_task = {"status": "thinking", "message": "正在理解..."}
    # 意图解析
    intent = parse_intent(req.text)  # 本地规则 + 可选云端

@app.get("/api/v1/status")
async def get_status():
    return current_task

@app.post("/api/v1/cancel")
async def cancel_task():
    executor.cancel()
    current_task = {"status": "idle", "message": "已取消"}
    return {"status": "cancelled"}

def run_task(intent):
    global current_task
    try:
        current_task = {"status": "executing", "message": f"正在{intent['action']}"}
        executor.execute(intent)
        current_task = {"status": "success", "message": "完成啦"}
    except Exception as e:
        current_task = {"status": "error", "message": str(e)}
```

#### 2.2.3 意图解析模块（`intent_parser.py`）

```python
import re
from local_intents import INTENT_TEMPLATES  # 预定义字典

def parse_intent(text: str) -> dict:
    # 1. 精确匹配动作缓存（SQLite）
    cached = query_action_cache(text)
    if cached:
        return {"source": "cache", "actions": cached}

    # 2. 规则匹配
    for template in INTENT_TEMPLATES:
        match = re.match(template["pattern"], text)
        if match:
            return {
                "source": "local",
                "action": template["action"],
                "slots": match.groupdict()
            }

    # 3. 云端解析（需网络）
    if is_network_available():
        result = call_autoglm(text)
        if result:
            return {"source": "cloud", "actions": result}
    
    # 4. 兜底
    return {"source": "fallback", "action": "unknown", "reply": "阿叔，我没听懂，能再说一遍吗？"}
```

#### 2.2.4 自动化执行引擎（`executor.py`）

```python
import time, json
from uiautomator2 import Device

class ActionExecutor:
    def __init__(self, device: Device):
        self.d = device
        self.cancel_flag = False

    def execute(self, intent: dict):
        actions = intent.get("actions", [])
        for step in actions:
            if self.cancel_flag:
                break
            self._do_step(step)

    def _do_step(self, step: dict):
        method = step["method"]
        params = step.get("params", {})
        # 执行前快照
        before = self.d.dump_hierarchy()
        if method == "click_text":
            self.d(text=params["text"]).click()
        elif method == "click_coord":
            x, y = params["x"], params["y"]
            self.d.click(x, y)
        elif method == "launch_app":
            self.d.app_start(params["package"])
        elif method == "input_text":
            self.d.send_keys(params["text"])
        # 等待界面变化
        time.sleep(1.5)
        after = self.d.dump_hierarchy()
        if before == after and method != "launch_app":
            # 无变化，重试
            self._retry(step, before)

    def _retry(self, step, before_xml):
        # 最多重试2次，每次先按返回键
        for i in range(2):
            self.d.press("back")
            time.sleep(1)
            self._do_step(step)
            after = self.d.dump_hierarchy()
            if after != before_xml:
                return
        # 上报云端
        screenshot = self.d.screenshot(format="opencv")
        desensitized = desensitize_screenshot(screenshot)
        upload_for_analysis(desensitized, before_xml)
        raise Exception("操作失败，界面无响应")
```

#### 2.2.5 隐私脱敏引擎（`privacy_engine.py`）

```python
import cv2
import numpy as np
from paddleocr import PaddleOCR

ocr = PaddleOCR(lang='ch', use_angle_cls=False, use_gpu=False)

SENSITIVE_PATTERNS = [
    r'\d{11}', r'\d{16,19}', r'\d{6}', r'\d+\.?\d{0,2}元?',  # 手机号、银行卡、验证码、金额
]

def desensitize_screenshot(img: np.ndarray) -> np.ndarray:
    result = ocr.ocr(img, cls=False)
    if not result or not result[0]:
        return img
    for line in result[0]:
        box, (text, conf) = line[0], line[1]
        for pattern in SENSITIVE_PATTERNS:
            if re.match(pattern, text):
                # 获取区域坐标并填充黑色
                pts = np.array(box, np.int32)
                x, y, w, h = cv2.boundingRect(pts)
                img[y:y+h, x:x+w] = 0
                break
    return img
```

## 五、适老化细节实现要点

1. **方言支持**：录制常用沪语 TTS 短语（“正在找囡囡”“稍等一歇”），通过本地音频播放，避免合成不自然。
2. **大字体与高对比度**：App 使用 24sp 以上字号，按钮最小触摸区域 48dp。
3. **动作回放优先**：第二次相同指令直接执行缓存，延迟 < 500ms，并语音提示“老快额，马上好”。
4. **误操作保护**：执行期间屏蔽系统导航栏（需辅助功能），仅保留悬浮停止按钮。

## 一、截图获取与预处理

### 1.1 截图获取（Termux 端）

```python
import uiautomator2 as u2
import cv2
import numpy as np
from PIL import Image
import io
import base64

d = u2.connect()

def capture_screen() -> np.ndarray:
    """
    获取当前屏幕截图，返回 OpenCV 格式的 numpy 数组（BGR）
    """
    # uiautomator2 直接返回 PIL Image 对象
    pil_img = d.screenshot(format="pillow")
    # 转换为 OpenCV 格式 (BGR)
    opencv_img = cv2.cvtColor(np.array(pil_img), cv2.COLOR_RGB2BGR)
    return opencv_img

def capture_screen_base64() -> str:
    """
    获取截图并转为 base64 字符串（用于上传云端）
    """
    pil_img = d.screenshot(format="pillow")
    buffer = io.BytesIO()
    pil_img.save(buffer, format="JPEG", quality=75)  # 压缩以减小传输体积
    return base64.b64encode(buffer.getvalue()).decode("utf-8")
```

### 1.2 获取当前界面元数据（辅助上下文）

截图本身缺乏语义信息，需同时提取 UI 树摘要作为 AutoGLM 的上下文：

```python
import xml.etree.ElementTree as ET

def get_ui_summary() -> dict:
    """
    提取当前界面的 UI 树摘要，包含可交互元素列表
    """
    xml_str = d.dump_hierarchy()
    root = ET.fromstring(xml_str)
    
    elements = []
    # 遍历所有可交互节点
    for node in root.iter():
        if node.attrib.get("clickable") == "true" or \
           node.attrib.get("long-clickable") == "true" or \
           node.attrib.get("scrollable") == "true":
            
            bounds = node.attrib.get("bounds", "")
            text = node.attrib.get("text", "")
            content_desc = node.attrib.get("content-desc", "")
            resource_id = node.attrib.get("resource-id", "")
            class_name = node.attrib.get("class", "")
            
            elements.append({
                "text": text,
                "content_desc": content_desc,
                "resource_id": resource_id.split("/")[-1] if resource_id else "",
                "class": class_name,
                "bounds": bounds  # 格式: "[x1,y1][x2,y2]"
            })
    
    # 获取当前 App 信息
    app_info = d.app_current()
    
    return {
        "app_package": app_info.get("package", ""),
        "app_activity": app_info.get("activity", ""),
        "interactive_elements": elements[:50]  # 限制数量，避免 token 过长
    }
```

## 二、隐私脱敏（上传前必须执行）

```python
import re
from paddleocr import PaddleOCR

# 初始化 PaddleOCR（轻量版，移动端可用）
ocr = PaddleOCR(lang='ch', use_angle_cls=False, use_gpu=False, 
                det_db_thresh=0.3, rec_batch_num=1)

# 敏感信息正则
SENSITIVE_PATTERNS = {
    "phone": re.compile(r'1[3-9]\d{9}'),
    "bank_card": re.compile(r'\d{16,19}'),
    "id_card": re.compile(r'\d{17}[\dXx]'),
    "money": re.compile(r'\d+\.?\d{0,2}\s*元'),
    "verification_code": re.compile(r'\b\d{4,6}\b'),
}

def desensitize_image(img: np.ndarray) -> np.ndarray:
    """
    对截图中的敏感信息进行黑块覆盖
    返回脱敏后的图像
    """
    img_copy = img.copy()
    h, w = img_copy.shape[:2]
    
    # OCR 识别
    results = ocr.ocr(img_copy, cls=False)
    if not results or not results[0]:
        return img_copy
    
    for line in results[0]:
        box = line[0]       # 四点坐标 [[x1,y1],[x2,y2],[x3,y3],[x4,y4]]
        text = line[1][0]   # 识别文本
        
        is_sensitive = False
        for pattern_name, pattern in SENSITIVE_PATTERNS.items():
            if pattern.search(text):
                is_sensitive = True
                break
        
        if is_sensitive:
            # 计算包围矩形
            pts = np.array(box, np.int32)
            x_min = max(0, int(pts[:, 0].min()))
            y_min = max(0, int(pts[:, 1].min()))
            x_max = min(w, int(pts[:, 0].max()))
            y_max = min(h, int(pts[:, 1].max()))
            # 黑块填充
            img_copy[y_min:y_max, x_min:x_max] = 0
    
    return img_copy

def prepare_upload_payload(img: np.ndarray, target_description: str) -> dict:
    """
    准备上传到云端 VLM 的完整载荷
    """
    # 脱敏
    clean_img = desensitize_image(img)
    
    # 转为 base64
    _, buffer = cv2.imencode('.jpg', clean_img, [cv2.IMWRITE_JPEG_QUALITY, 75])
    img_base64 = base64.b64encode(buffer).decode('utf-8')
    
    # 获取 UI 摘要
    ui_summary = get_ui_summary()
    
    return {
        "image_base64": img_base64,
        "target": target_description,           # 如 "视频通话按钮"
        "ui_summary": ui_summary,
        "device_resolution": {
            "width": img.shape[1],
            "height": img.shape[0]
        }
    }
```

```python
    img = capture_screen()
    
    # 准备载荷
    payload = prepare_upload_payload(img, target)
    
    for attempt in range(max_retries):
        try:
            response = requests.post(
                CLOUD_VLM_URL,
                json=payload,
                timeout=15,
                headers={"Authorization": "Bearer YOUR_API_KEY"}
            )
            response.raise_for_status()
            result = response.json()
            
            # 验证返回格式
            if "x" in result and "y" in result:
                return {
                    "x": float(result["x"]),       # 归一化 x (0.0 ~ 1.0)
                    "y": float(result["y"]),       # 归一化 y (0.0 ~ 1.0)
                    "confidence": result.get("confidence", 0.0),
                    "action": result.get("action", "click"),
                    "description": result.get("description", "")
                }
            else:
                raise ValueError(f"VLM 返回格式异常: {result}")
                
        except Exception as e:
            if attempt == max_retries - 1:
                raise Exception(f"VLM 定位失败，已重试 {max_retries} 次: {str(e)}")
            continue
```

```python
import time

def execute_click(x: float, y: float, action_type: str = "click") -> bool:
    """
    执行点击操作
    支持单击、长按、双击
    """
    screen_w, screen_h = get_screen_resolution()
    actual_x, actual_y = denormalize_coord(x, y, screen_w, screen_h)
    
    # 边界保护
    actual_x = max(1, min(screen_w - 1, actual_x))
    actual_y = max(1, min(screen_h - 1, actual_y))
    
    if action_type == "click":
        d.click(actual_x, actual_y)
    elif action_type == "long_press":
        d.long_click(actual_x, actual_y, duration=1.0)
    elif action_type == "double_click":
        d.double_click(actual_x, actual_y)
    else:
        d.click(actual_x, actual_y)
    
    return True

def execute_with_verification(x: float, y: float, 
                               expected_change: bool = True,
                               timeout: float = 3.0) -> bool:
    """
    执行点击并验证界面是否发生变化
    """
    # 点击前快照
    before_xml = d.dump_hierarchy()
    
    # 执行点击
    execute_click(x, y)
    
    # 等待界面响应
    start_time = time.time()
    while time.time() - start_time < timeout:
        time.sleep(0.5)
        after_xml = d.dump_hierarchy()
        
        if expected_change and after_xml != before_xml:
            return True  # 界面已变化，点击生效
        elif not expected_change:
            return True
    
    return False  # 超时未变化
```

## 五、多策略点击决策引擎

实际场景中，不应仅依赖 VLM 坐标，需组合多种策略以提高成功率：

```python
class ClickStrategy:
    """
    多策略点击引擎
    按优先级依次尝试：文本匹配 → 坐标点击 → 图像模板匹配
    """
    
    def __init__(self):
        self.d = u2.connect()
    
    def smart_click(self, target: str, vlm_coord: dict = None) -> bool:
        """
        智能点击，自动选择最优策略
        """
        # 策略1：UI 树文本/描述匹配（最快最准）
        if self._click_by_text(target):
            return True
        
        # 策略2：UI 树 content-desc 匹配
        if self._click_by_desc(target):
            return True
        
        # 策略3：VLM 返回的坐标点击
        if vlm_coord and vlm_coord.get("confidence", 0) > 0.7:
            if self._click_by_coord(vlm_coord["x"], vlm_coord["y"]):
                return True

        if self._click_by_template(target):

        if vlm_coord:

        return False
```

### 5.1 本地图像模板匹配（离线备选方案）

当网络不可用时，使用预存的图标模板进行匹配：

```python
import cv2
import os

TEMPLATE_DIR = os.path.expanduser("~/.hermes/templates/")

def match_template_on_screen(template_name: str, threshold: float = 0.8) -> tuple:
    """
    在屏幕上查找预存模板图像的位置
    返回 (x, y) 或 None
    """
    template_path = os.path.join(TEMPLATE_DIR, f"{template_name}.png")
    if not os.path.exists(template_path):
        return None
    
    # 获取屏幕截图
    screen = capture_screen()
    screen_gray = cv2.cvtColor(screen, cv2.COLOR_BGR2GRAY)
    
    # 加载模板
    template = cv2.imread(template_path, cv2.IMREAD_GRAYSCALE)
    
    # 多尺度匹配（应对不同分辨率）
    scales = [0.8, 0.9, 1.0, 1.1, 1.2]
    best_match = None
    best_val = 0
    
    for scale in scales:
        resized = cv2.resize(template, None, fx=scale, fy=scale)
        if resized.shape[0] > screen_gray.shape[0] or \
           resized.shape[1] > screen_gray.shape[1]:
            continue
        
        result = cv2.matchTemplate(screen_gray, resized, cv2.TM_CCOEFF_NORMED)
        min_val, max_val, min_loc, max_loc = cv2.minMaxLoc(result)
        
        if max_val > best_val:
            best_val = max_val
            h, w = resized.shape
            best_match = (max_loc[0] + w // 2, max_loc[1] + h // 2)
    
    if best_val >= threshold and best_match:
        return best_match
    
    return None
```

## 六、完整的端到端执行流程

```python
class HermesVisualExecutor:
    """
    完整的视觉定位 + 执行流水线
    """
    
    def __init__(self):
        self.d = u2.connect()
        self.clicker = ClickStrategy()
        self.screen_w, self.screen_h = get_screen_resolution()
    
    def execute_task(self, task_description: str, target_element: str) -> dict:
        """
        执行一个需要视觉定位的任务
        返回执行结果
        """
        result = {
            "success": False,
            "method": "",
            "attempts": 0,
            "error": None
        }
        
        for attempt in range(3):  # 最多尝试 3 次
            result["attempts"] = attempt + 1
            
            try:
                # 1. 尝试本地策略（无需网络）
                if self.clicker.smart_click(target_element, vlm_coord=None):
                    result["success"] = True
                    result["method"] = "local_text_match"
                    break
                
                # 2. 本地失败，请求云端 VLM
                vlm_result = request_visual_location(target_element)
                
                if vlm_result["action"] == "click":
                    # 3. 执行 VLM 返回的坐标
                    if self.clicker.smart_click(target_element, vlm_coord=vlm_result):
                        result["success"] = True
                        result["method"] = f"vlm_coord (conf={vlm_result['confidence']:.2f})"
                        break
                
                elif vlm_result["action"] == "scroll_down":
                    # 需要滚动
                    self.d.swipe(
                        self.screen_w // 2, self.screen_h * 3 // 4,
                        self.screen_w // 2, self.screen_h // 4,
                        duration=0.3
                    )
                    time.sleep(1.0)
                    continue  # 滚动后重试
                
                elif vlm_result["action"] == "not_found":
                    result["error"] = f"未找到目标: {vlm_result.get('reason', '未知原因')}"
                    break
                    
            except Exception as e:
                result["error"] = str(e)
                if attempt < 2:
                    time.sleep(1.5)
                    continue
        
        return result
```

当 uiautomator2 的 ADB 方式受限时（如部分手机封锁无线调试），可降级使用 Android 辅助功能服务：

```kotlin
// Android 端 AccessibilityService
class HermesAccessibilityService : AccessibilityService() {
    
    override fun onAccessibilityEvent(event: AccessibilityEvent?) {}
    
    override fun onInterrupt() {}
    
    /**
     * 接收来自 Termux 的点击指令（通过 ContentProvider 或 LocalBroadcast）
     */
    fun performClick(x: Float, y: Float) {
        val path = Path().apply {
            moveTo(x, y)
        }
        val gesture = GestureDescription.Builder()
            .addStroke(GestureDescription.StrokeDescription(path, 0, 100))
            .build()
        dispatchGesture(gesture, null, null)
    }
    
    /**
     * 通过 AccessibilityNodeInfo 查找并点击文本
     */
    fun clickByText(text: String): Boolean {
        val root = rootInActiveWindow ?: return false
        val nodes = root.findAccessibilityNodeInfosByText(text)
        for (node in nodes) {
            if (node.isClickable) {
                node.performAction(AccessibilityNodeInfo.ACTION_CLICK)
                return true
            }
        }
        return false
    }
}
```
